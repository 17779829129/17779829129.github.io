#include <REGX52.H>

void Delay1ms(unsigned int xms)        
{
    unsigned char i, j;
    while(xms)
    {
        i = 2;
        j = 239;
        do
        {
            while (--j);
        } while (--i);
        xms--;
    }
}

void main()
{
    unsigned char num=0;
    while(1)
    {
        P2=~(0x01<<num);
        Delay1ms(500);
        num++;
        if(num>=8)
            num=0;
    }
}

#include <REGX52.H>

    unsigned char data i, j;

{
    unsigned char num=0;

    while(1)

            num++;
            if(num>=8)

                num=0;

                    num=7;
                else
                    num--;
                P2=~(0x01<<num);

    }

位运算

运算符:
<< :按位左移
0011 1100<<1 -> 0111 1000

>> :按位右移
0011 1100<<2 -> 0000 1111

& :按位与
0001 1000 & 0010 1010 -> 0000 1000

| :按位或
0001 1000 | 0010 1010 -> 0011 1010

^ :按位异或
0001 1000 ^ 0010 1010 -> 0011 0010
~ : 按位取反
~0001 1000 -> 1110 0111

矩阵键盘密码锁

代码：

#include <REGX52.H>
#include "LCD1602.h"
#include "Delay.h"
#include "MatrixKey.h"

unsigned char num=0;
unsigned int password=0,count=0;
void main()
{
    LCD_Init();
    LCD_ShowString(1,1,"password:");
    while(1)
    {
        num=keynumber();
        if(num)
        {
            if(num<=10)
            {
                if(count<4)
                {
                    password*=10;
                    password+=num%10;
                    count++;
                }
                LCD_ShowNum(2,1,password,4);
            }
            if(num==11)
            {
                if(password==2345)
                {
                    LCD_ShowString(1,14,"OK ");
                    password=0;
                    count=0;
                    LCD_ShowNum(2,1,password,4);
                }
                else 
                {
                    LCD_ShowString(1,14,"err");
                    password=0;
                    count=0;
                    LCD_ShowNum(2,1,password,4);
                }
            }
            if(num==12)
            {
                password/=10;
                count--;
                LCD_ShowNum(2,1,password,4);
            }
        }
    }
}
